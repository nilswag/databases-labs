-- DDL for new tables
CREATE TABLE ConcertType (
    TypeId INTEGER,
    Title TEXT NOT NULL,
    PRIMARY KEY (TypeID)
);

CREATE TABLE Concert (
    ConcertId INTEGER,
    TypeID INTEGER NOT NULL,
    Date TEXT,
    StartTime TEXT,
    Duration REAL,
    Location TEXT NOT NULL,
    Capacity INTEGER NOT NULL,
    PRIMARY KEY (ConcertId),
    FOREIGN KEY (TypeID) REFERENCES ConcertType(TypeId)
);

CREATE TABLE Ticket (
    TicketId INTEGER,
    ConcertId INTEGER NOT NULL,
    Price Integer NOT NULL,
    PRIMARY KEY (TicketId),
    FOREIGN KEY (ConcertId) REFERENCES Concert(ConcertId)
);

CREATE TABLE SittingTicket (
    TicketId INTEGER,
    SeatNr INTEGER NOT NULL,
    PRIMARY KEY (TicketId),
    FOREIGN KEY (TicketId) REFERENCES Ticket(TicketId)
);

CREATE TABLE StandingTicket (
    TicketId INTEGER,
    RegionNr INTEGER NOT NULL,
    PRIMARY KEY (TicketId),
    FOREIGN KEY (TicketId) REFERENCES Ticket(TicketId)
);

CREATE TABLE Attendee (
    AttendeeId INTEGER,
    TicketId INTEGER NOT NULL,
    Name TEXT NOT NULL,
    Address TEXT NOT NULL,
    PRIMARY KEY (AttendeeId),
    FOREIGN KEY (TicketId) REFERENCES Ticket(TicketId)
);

CREATE TABLE Buyer (
    AttendeeId INTEGER,
    EMail TEXT NOT NULL,
    Password TEXT NOT NULL,
    PaymentMethod TEXT NOT NULL,
    PRIMARY KEY (AttendeeId),
    FOREIGN KEY (AttendeeId) REFERENCES Attendee(AttendeeId)
);

CREATE TABLE Employee (
    EmployeeId INTEGER,
    ConcertId INTEGER,
    Name TEXT NOT NULL,
    Address TEXT NOT NULL,
    Salary INTEGER NOT NULL,
    PRIMARY KEY (EmployeeId),
    FOREIGN KEY (ConcertId) REFERENCES Concert(ConcertId)
);

CREATE TABLE Artist(
    EmployeeId INTEGER,
    Genre TEXT,
    Popularity REAL,
    PRIMARY KEY (EmployeeId),
    FOREIGN KEY (EmployeeId) REFERENCES Employee(EmployeeId)
);

CREATE TABLE SecurityGuard (
    EmployeeId INTEGER,
    Rank TEXT NOT NULL,
    PRIMARY KEY (EmployeeId),
    FOREIGN KEY (EmployeeId) REFERENCES Employee(EmployeeId)
);

CREATE TABLE Organizer (
    EmployeeId INTEGER,
    Specialization TEXT NOT NULL,
    PRIMARY KEY (EmployeeId),
    FOREIGN KEY (EmployeeId) REFERENCES Employee(EmployeeId)
);
